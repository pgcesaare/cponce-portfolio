```sql
SELECT 
    ranch_id,
    COUNT(cattle_id) AS total_heads,
    SUM(weight) AS total_weight,
    AVG(feeding_days) AS avg_feeding_days
FROM cattle_inventory
WHERE status = 'Active'
GROUP BY ranch_id
ORDER BY total_heads DESC;
```
